2. Write a PL/SQL block to display the letter grade(0-4: F; 4-5: E; 5-6: D; 6-7: C;               
7-8: B; 8-9: A; 9-10: A+} of given student.  
DECLARE
ROLL StudentTable.RollNo%TYPE;
GPA_I StudentTable.GPA%TYPE;
GRADE CHAR(2);
BEGIN
ROLL:=&RollNo;
SELECT GPA INTO GPA_I FROM StudentTable WHERE RollNo=ROLL;
IF GPA_I >= 0 AND GPA_I < 4 THEN
GRADE := 'F';
ELSIF GPA_I >= 4 AND GPA_I < 5 THEN
GRADE := 'E';
ELSIF GPA_I >= 5 AND GPA_I < 6 THEN
GRADE := 'D';
ELSIF GPA_I >= 6 AND GPA_I < 7 THEN
GRADE := 'C';
ELSIF GPA_I >= 7 AND GPA_I < 8 THEN
GRADE := 'B';
ELSIF GPA_I >= 8 AND GPA_I < 9 THEN
GRADE := 'A';
ELSIF GPA_I >= 9 AND GPA_I <= 10 THEN
GRADE := 'A+';
ELSE  
DBMS_OUTPUT.PUT_LINE('NO SUCH GRADE FOR THE GPA');
END IF;
DBMS_OUTPUT.PUT_LINE('GRADE OF THE STUDENT WITH ROLL NO. ' || TO_CHAR(ROLL) || ' AND ' || TO_CHAR(GPA_I) || ' IS ' || GRADE);
END;
/

DECLARE
ISSUED DATE;
RETURND DATE;
FINE NUMBER;
DAYS NUMBER;
BEGIN
ISSUED := TO_DATE('&ISSUE_DATE', 'DD-MM-YYYY');
RETURND := TO_DATE('&RETURN_DATE', 'DD-MM-YYYY');
DAYS:=RETURND-ISSUED;

END IF;
IF DAYS<=7 THEN
FINE:=0;
ELSIF DAYS>=8 AND DAYS<=15 THEN
FINE:= DAYS*1;
ELSIF DAYS>=16 AND DAYS<=30 THEN
FINE:= 8 + (DAYS-15)*2;

FINE:=(8*1) + (15*2) + (DAYS-30)*5;

4. Write a PL/SQL block to print the letter grade of all the students(RollNo: 1 - 5). 
SET SERVEROUTPUT ON;

DECLARE
    V_ROLL  NUMBER := 1;
    V_GPA   StudentTable.GPA%TYPE;
    V_GRADE VARCHAR2(2);
BEGIN
    LOOP
        EXIT WHEN V_ROLL > 5;

        SELECT GPA INTO V_GPA
        FROM StudentTable
        WHERE RollNo = V_ROLL;

        IF V_GPA < 4 THEN
            V_GRADE := 'F';
        ELSIF V_GPA < 5 THEN
            V_GRADE := 'E';
        ELSIF V_GPA < 6 THEN
            V_GRADE := 'D';
        ELSIF V_GPA < 7 THEN
            V_GRADE := 'C';
        ELSIF V_GPA < 8 THEN
            V_GRADE := 'B';
        ELSIF V_GPA < 9 THEN
            V_GRADE := 'A';
        ELSE
            V_GRADE := 'A+';
        END IF;

        DBMS_OUTPUT.PUT_LINE('Roll: ' || V_ROLL || 
                             ' GPA: ' || V_GPA || 
                             ' Grade: ' || V_GRADE);

        V_ROLL := V_ROLL + 1;
    END LOOP;
END;
/

6. Write a PL/SQL block to find the student with max. GPA without using aggregate 
function.  
DECLARE
    V_MAX_GPA StudentTable.GPA%TYPE := 0;
    V_MAX_ROLL NUMBER := 0;
BEGIN
    FOR REC IN (SELECT RollNo, GPA FROM StudentTable) LOOP
        IF REC.GPA > V_MAX_GPA THEN
            V_MAX_GPA := REC.GPA;
            V_MAX_ROLL := REC.RollNo;
        END IF;
    END LOOP;

    DBMS_OUTPUT.PUT_LINE('Top Student Roll: ' || V_MAX_ROLL ||
                         ' GPA: ' || V_MAX_GPA);
END;

8. Based on the University database schema, write a PL/SQL block to display the 
details of the Instructor whose name is supplied by the user. Use exceptions to 
show appropriate error message for the following cases:  
a. Multiple instructors with the same name  
b. No instructor for the given name   
SET SERVEROUTPUT ON;

DECLARE
    V_NAME  instructor.name%TYPE := '&INSTRUCTOR_NAME';
    V_ID    instructor.ID%TYPE;
    V_DEPT  instructor.dept_name%TYPE;
    V_SAL   instructor.salary%TYPE;
BEGIN
    SELECT ID, dept_name, salary
    INTO V_ID, V_DEPT, V_SAL
    FROM instructor
    WHERE name = V_NAME;

    DBMS_OUTPUT.PUT_LINE('ID: ' || V_ID);
    DBMS_OUTPUT.PUT_LINE('Dept: ' || V_DEPT);
    DBMS_OUTPUT.PUT_LINE('Salary: ' || V_SAL);

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('No instructor found with this name.');

    WHEN TOO_MANY_ROWS THEN
        DBMS_OUTPUT.PUT_LINE('Multiple instructors exist with this name.');

    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Unexpected error occurred.');
END;
/
